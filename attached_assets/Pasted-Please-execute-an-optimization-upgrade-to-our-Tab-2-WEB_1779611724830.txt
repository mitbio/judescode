      const events = response.data.events || [];

        const creationYear = new Date(targetEvent.eventDate).getFullYear();
        const currentYear = 2026; // Sync with platform operational calendar year
        const age = currentYear - creationYear;
        return age === 0 ? "New Brand (<1 yr)" : `${age} yrs old (Est. ${creationYear})`;
      }
      return "Established Brand";
    } catch (err) {
      return "Active Domain"; // Clean, non-crashing fallback UI metric
    }
  }
