# Initialize Session State (The App's Memory)
if "has_run" not in st.session_state:
    st.session_state.has_run = False
if "generated_ideas" not in st.session_state:
    st.session_state.generated_ideas = []
if "idea_counter" not in st.session_state:
    st.session_state.idea_counter = 0

# --- Callbacks ---
def trigger_run():
    st.session_state.has_run = True
    st.session_state.generated_ideas = [] # Reset on new run
    st.session_state.idea_counter = 0
    generate_new_batch()

def generate_new_batch():
    # A mock engine to generate 2 ideas at a time
    hooks = [
        "Stop selling the mansion. Start selling the silence.",
        "The Open Floor Plan is Dead. Here's What Killed It.",
        "Your clients aren't buying real estate. They are buying a fortress.",
        "Why 'Boring' Architecture is the new ultimate flex in 2026.",
        "The highest ROI room in a luxury home isn't the kitchen anymore."
    ]
    
    for _ in range(2):
        st.session_state.idea_counter += 1
        new_idea = {
            "id": f"{st.session_state.idea_counter:02d}",
            "hook": random.choice(hooks),
            "angle": "Contrarian breakdown: Most agents focus on amenities. We focus on psychological safety and spatial compartmentalization to drive urgency.",
            "visual": "Cinematic 35mm shot, high contrast, moody lighting. Minimalist composition."
        }
        st.session_state.generated_ideas.append(new_idea)

# Callbacks for demo interactions
def push_to_notion(idea_id):
    st.toast(f"✅ Idea {idea_id} successfully synced to Notion Vault.", icon="⚡")

def reroll_hook(idea_id):
    st.toast(f"🔄 Rerolling hook for Angle {idea_id} via LLM...", icon="🧠")

def generate_visual(idea_id):
    st.toast(f"🎨 Nano Banana 2 is generating visual concept for Angle {idea_id}...", icon="✨")
